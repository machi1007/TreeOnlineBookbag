    #include <stdio.h>

// 主函數
int main() {
    // 變量宣告
    int a = 10;
    float b = 20.5;
    char c = 'A';

    // 條件語句
    if (a > 5) {
        printf("a is greater than 5\n");
    } else {
        printf("a is not greater than 5\n");
    }

    // 循環
    for(int i = 0; i < a; i++) {
        printf("%d ", i);
    }
    printf("\n");

    // while 循環
    int count = 0;
    while (count < a) {
        printf("%d ", count);
        count++;
    }
    printf("\n");

    // 函數調用
    int result = add(a, b);
    printf("Result: %d\n", result);

    // 指針
    int* ptr = &a;
    printf("Address of a: %p\n", (void*)ptr);
    printf("Value of a through pointer: %d\n", *ptr);

    return 0;
}

// 函數定義
int add(int x, int y) {
    return x + y;
}

  • C#語言（2024.04.02）

     
    using System;

namespace HelloWorld
{
    class Program
    {
        static void Main(string[] args)
        {
            // 變量宣告
            int a = 10;
            double b = 20.5;
            string greeting = "Hello, World!";
            
            // 輸出到控制台
            Console.WriteLine(greeting);
            
            // 條件語句
            if (a > 5)
            {
                Console.WriteLine("a is greater than 5");
            }
            else
            {
                Console.WriteLine("a is not greater than 5");
            }
            
            // 循環
            for(int i = 0; i < 5; i++)
            {
                Console.WriteLine("i = " + i);
            }
            
            // 方法調用
            int sum = Add(a, a);
            Console.WriteLine("Sum = " + sum);
            
            // 創建對象
            MyClass obj = new MyClass();
            obj.MyMethod();
        }
        
        // 方法定義
        static int Add(int x, int y)
        {
            return x + y;
        }
    }
    
    // 類別定義
    class MyClass
    {
        public void MyMethod()
        {
            Console.WriteLine("In MyMethod of MyClass");
        }
    }
    
    // 繼承
    class DerivedClass : MyClass
    {
        public void AnotherMethod()
        {
            Console.WriteLine("In AnotherMethod of DerivedClass");
        }
    }
}
